-- 1. Limpeza de tabelas antigas de teste
DROP TABLE IF EXISTS public.game_configs;
DROP TABLE IF EXISTS public.characters;
DROP TABLE IF EXISTS public.accounts;

-- 2. Tabela de Contas
CREATE TABLE public.accounts (
id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
account_name TEXT NOT NULL UNIQUE,
password TEXT NOT NULL, -- Texto puro para o MVP
created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- 3. Tabela de Personagens (Comercial, Performática e Sem Duplicidade)
CREATE TABLE public.characters (
id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
account_id UUID NOT NULL REFERENCES public.accounts(id) ON DELETE CASCADE,

    -- Dados de Identificação e Busca Rápida
    name TEXT NOT NULL UNIQUE,
    name_search TEXT NOT NULL UNIQUE, -- Sempre minúsculo e sem espaços
    sex TEXT NOT NULL CHECK (sex IN ('male', 'female')),
    level INTEGER DEFAULT 1 NOT NULL,
    vocation TEXT DEFAULT 'No Vocation' NOT NULL,
    experience BIGINT DEFAULT 0 NOT NULL,
    is_admin BOOLEAN DEFAULT false NOT NULL,
    
    -- Coordenadas Gerenciadas pelo Backend no Mapa
    pos_x INTEGER DEFAULT 82 NOT NULL,
    pos_y INTEGER DEFAULT 81 NOT NULL,
    pos_z INTEGER DEFAULT 8 NOT NULL,
    
    -- Posição do Templo (Respawn)
    temple_x INTEGER DEFAULT 82 NOT NULL,
    temple_y INTEGER DEFAULT 81 NOT NULL,
    temple_z INTEGER DEFAULT 8 NOT NULL,

    -- JSONB de Inventário e Status Dinâmicos de RPG (health, equipment, spellbook, amigos e os contadores de achievements)
    game_data JSONB NOT NULL, 

    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- 4. Tabela de Configurações da Engine (Onde guardamos as Enums e as Definições de Achievements)
CREATE TABLE public.game_configs (
key TEXT PRIMARY KEY,
data JSONB NOT NULL,
updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);
